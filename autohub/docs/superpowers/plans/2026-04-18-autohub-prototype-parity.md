> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Port visual parity (hero, landing, features page) and close functional gaps (RBAC schema, password reset, webhook proxy with async/HMAC, tool lifecycle, admin componentization, per-user access grants) from the prototype into the autohub monorepo.

**Architecture:** Schema-first approach — new Drizzle migration establishes all new tables/columns, then API routes are added/updated, then frontend. Each track (visual / functional) is sequenced so tests can run at each step. The existing `ToolExecutionService` is preserved and extended by a new `WebhookProxyService` that wraps it.

**Tech Stack:** Next.js 15 App Router (web), Hono + Drizzle ORM + PostgreSQL (api), `@autohub/shared` (types/validators), shadcn-ui + Tailwind (UI), bcryptjs + jsonwebtoken (auth), Node.js `crypto` module (HMAC signing), Playwright (e2e), Vitest (unit/integration).

import { createHmac, timingSafeEqual } from "crypto";

export function signPayload(secret: string, timestamp: string, executionId: string, rawBody: string): string {
  const message = `${timestamp}.${executionId}.${rawBody}`;
  return "sha256=" + createHmac("sha256", secret).update(message).digest("hex");
}

interface VerifyParams {
  secret: string;
  timestamp: string;
  executionId: string;
  rawBody: string;
  signature: string;
}

export function verifySignature({ secret, timestamp, executionId, rawBody, signature }: VerifyParams): boolean {
  const now = Math.floor(Date.now() / 1000);
  const ts = parseInt(timestamp, 10);
  if (Math.abs(now - ts) > 300) return false;

  const expected = signPayload(secret, timestamp, executionId, rawBody);
  try {
    return timingSafeEqual(Buffer.from(expected), Buffer.from(signature));
  } catch {
    return false;
  }
}
```

import { createMiddleware } from "hono/factory";

const ROLE_HIERARCHY: Record<string, number> = { user: 0, moderator: 1, admin: 2 };

export function requireRole(minRole: "moderator" | "admin") {
  return createMiddleware(async (c, next) => {
    const user = c.get("user");
    if (!user) return c.json({ error: "Unauthorized" }, 401);
    const userLevel = ROLE_HIERARCHY[user.role] ?? 0;
    const required = ROLE_HIERARCHY[minRole];
    if (userLevel < required) return c.json({ error: "Forbidden" }, 403);
    await next();
  });
}
```

  const { id } = c.req.param();

toolsRouter.patch("/:id/status", requireAuth, requireRole("admin"), async (c) => {
  const { id } = c.req.param();
  const body = await c.req.json<{ status: "approved" | "rejected" | "archived"; reason?: string }>();
  if (!["approved", "rejected", "archived"].includes(body.status)) {

    updatedAt: new Date(),

import { signPayload } from "./hmac.js";

const API_BASE_URL = process.env.AUTOHUB_API_URL ?? "http://localhost:4000";

export function generateSigningSecret(): string {
  return randomBytes(32).toString("hex");
}

interface ProxyParams {
  toolId: string;
  userId: string;
  userRole?: string;
  inputs: Record<string, unknown>;

export class WebhookProxyService {
  static async execute({ toolId, userId, userRole, inputs, ip }: ProxyParams) {
    const isAdmin = userRole === "admin";

    const [tool] = await db.select().from(aiTools).where(eq(aiTools.id, toolId)).limit(1);
    if (!tool) throw Object.assign(new Error("Tool not found"), { status: 404 });
    if (!tool.isActive || tool.approvalStatus !== "approved") {
      throw Object.assign(new Error("Tool not available"), { status: 400 });
    }
    if (!tool.webhookUrl) throw Object.assign(new Error("Tool has no webhook configured"), { status: 400 });

    // Credit pre-check for non-admins
    if (!isAdmin) {
      const [creditRow] = await db.select().from(credits).where(eq(credits.userId, userId)).limit(1);
      if (!creditRow || creditRow.currentCredits < tool.creditCost) {
        throw Object.assign(new Error("Insufficient credits"), { status: 402 });
      }
    }

    const executionId = randomBytes(16).toString("hex");
    const [execution] = await db.insert(executions).values({
      id: executionId,
      toolId,
      userId,
      status: "pending",
      requestPayload: inputs,
      creditsDebited: 0,
    }).returning();

    if (tool.executionMode === "sync") {
      return this.executeSync({ tool, execution, inputs, isAdmin, ip });
    } else {
      return this.executeAsync({ tool, execution, inputs, isAdmin });
    }
  }

  private static async executeSync({ tool, execution, inputs, isAdmin, ip }: {
    tool: typeof aiTools.$inferSelect;
    execution: typeof executions.$inferSelect;
    inputs: Record<string, unknown>;
    isAdmin: boolean;
    ip?: string;
  }) {
    const timestamp = Math.floor(Date.now() / 1000).toString();
    const body = JSON.stringify({ executionId: execution.id, inputs });
    const signature = tool.signingSecretHash
      ? signPayload(tool.signingSecretHash, timestamp, execution.id, body)
      : undefined;

    const headers: Record<string, string> = { "Content-Type": "application/json" };
    if (signature) {
      headers["X-AutoHub-Timestamp"] = timestamp;
      headers["X-AutoHub-Signature"] = signature;
    }

    const controller = new AbortController();
    const timeout = setTimeout(() => controller.abort(), (tool.webhookTimeout ?? 30) * 1000);

    try {
      const res = await fetch(tool.webhookUrl!, {
        method: "POST",
        headers,
        body,
        signal: controller.signal,
      }).finally(() => clearTimeout(timeout));

      if (!res.ok) {
        await db.update(executions)
          .set({ status: "failed", error: `Webhook returned ${res.status}`, completedAt: new Date() })
          .where(eq(executions.id, execution.id));
        return { executionId: execution.id, status: "failed" as const, error: `Webhook returned ${res.status}` };
      }

      const responsePayload = await res.json().catch(() => null);
      const creditsDebited = isAdmin ? 0 : tool.creditCost;

      if (!isAdmin) {
        await db.execute(
          import("drizzle-orm").then(({ sql }) =>
            sql`UPDATE credits SET current_credits = current_credits - ${tool.creditCost} WHERE user_id = ${execution.userId} AND current_credits >= ${tool.creditCost}`
          )
        );
      }

      await db.update(executions)
        .set({ status: "success", responsePayload, creditsDebited, completedAt: new Date() })
        .where(eq(executions.id, execution.id));

      return { executionId: execution.id, status: "success" as const, output: responsePayload, creditsDebited };
    } catch (err) {
      const isTimeout = (err as Error).name === "AbortError";
      await db.update(executions)
        .set({ status: isTimeout ? "timeout" : "failed", error: (err as Error).message, completedAt: new Date() })
        .where(eq(executions.id, execution.id));
      return { executionId: execution.id, status: isTimeout ? "timeout" as const : "failed" as const };
    }
  }

  private static async executeAsync({ tool, execution, inputs, isAdmin }: {
    tool: typeof aiTools.$inferSelect;
    execution: typeof executions.$inferSelect;
    inputs: Record<string, unknown>;
    isAdmin: boolean;
  }) {
    const callbackUrl = `${API_BASE_URL}/api/executions/${execution.id}/callback`;
    const timestamp = Math.floor(Date.now() / 1000).toString();
    const body = JSON.stringify({ executionId: execution.id, inputs, callbackUrl });
    const signature = tool.signingSecretHash
      ? signPayload(tool.signingSecretHash, timestamp, execution.id, body)
      : undefined;

    const headers: Record<string, string> = { "Content-Type": "application/json" };
    if (signature) {
      headers["X-AutoHub-Timestamp"] = timestamp;
      headers["X-AutoHub-Signature"] = signature;
    }

    // Fire and forget — don't await
    fetch(tool.webhookUrl!, { method: "POST", headers, body }).catch(() => {
      db.update(executions)
        .set({ status: "failed", error: "Failed to reach webhook", completedAt: new Date() })
        .where(eq(executions.id, execution.id));
    });

    return { executionId: execution.id, status: "pending" as const };
  }
}

  const { id } = c.req.param();

  const timestamp = c.req.header("x-autohub-timestamp") ?? "";
  const signature = c.req.header("x-autohub-signature") ?? "";

  if (!execution) return c.json({ error: "Not found" }, 404);

  if (["success", "failed", "timeout"].includes(execution.status)) {
    return c.json({ ok: true }, 200);

      secret: tool.signingSecretHash,
      timestamp,
      executionId: id,
      rawBody,
      signature,
    });
    if (!valid) return c.json({ error: "Invalid signature" }, 401);
  }

  const payload = JSON.parse(rawBody) as { success?: boolean; output?: unknown; error?: string };
  const isSuccess = payload.success !== false;
  const isAdmin = false; // callbacks don't have user context; check execution record

  if (isSuccess) {
    // Debit credits if not admin execution (creditsDebited == 0 and userId has role check)
    const [userRole] = await db.execute(
      sql`SELECT role FROM user_roles WHERE user_id = ${execution.userId} LIMIT 1`

      await db.execute(

        responsePayload: payload.output ?? payload,
        creditsDebited: shouldDebit ? tool.creditCost : 0,
        completedAt: new Date(),

    .select()
    .from(passwordResetTokens)
    .where(and(
      eq(passwordResetTokens.usedAt, null as unknown as Date),
      // expired tokens: expiresAt > now handled below
    ));

  let matched: typeof validTokens[0] | null = null;
  for (const row of validTokens) {
    if (row.expiresAt < now) continue;
    const ok = await bcrypt.compare(body.token, row.tokenHash);
    if (ok) { matched = row; break; }
  }

  await db.update(users).set({ passwordHash: newHash, updatedAt: new Date() }).where(eq(users.id, matched.userId));

  const [busy, setBusy] = useState<string | null>(null);

    if (!session?.apiToken) return;
    setBusy(tool.id);
    try {
      const res = await apiClient.patch<{ data: AITool }>(
        `/api/admin/tools/${tool.id}`,
        { isActive: !tool.isActive },

      setBusy(null);

      setBusy(null);
